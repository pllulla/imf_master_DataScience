#Preparamos una lista vac√≠a
dataframes = []

#Iterar sobre archivos CSV y cargar cada uno en un DataFrame separado utilizando la funci√≥n read_csv
#de Pandas. Se puede utilizar la funci√≥n glob de la biblioteca glob para obtener una lista de los
#archivos CSV en un directorio determinado:

import glob

#Ruta a los archivos CSV
ruta = "CSVs/*.csv"

#Itera sobre los archivos CSV, a tener en cuenta el separador y la codificaci√≥n de caracteres "latinos"
for archivo_csv in glob.glob(ruta):
    df = pd.read_csv(archivo_csv, sep= ";", encoding= "latin1")
    dataframes.append(df)
